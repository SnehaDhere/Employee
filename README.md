# Employee
public class Employee
{
    int emp_id;
    String emp_name;
    int emp_sal;

    void setEmp_id(int emp_id)
    {
        this.emp_id=emp_id;

    }
    int getEmp_id()
    {
        return emp_id;
    }
    void setEmp_name(String emp_name)
    {
        this.emp_name=emp_name;
    }
    String getEmp_name()
    {
        return emp_name;
    }
    void setEmp_sal(int emp_sal)
    {
        this.emp_sal=emp_sal;
    }
    int getEmp_sal()
    {
        return emp_sal;
    }
}
